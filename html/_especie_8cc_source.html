40 double Especie::distancia(const Especie& e) const
41 {
42  vector<string> kmer_p_i = this->kmer;
43  vector<string> kmer_e = e.kmer;
44  int s1 = kmer_p_i.size();
45  int s2 = kmer_e.size();
46  int interseccion = 0; //variable que cuenta el cardinal de la intersección de
47  // kmer_p_i y kmer_e
48  int i = 0;
49  int j = 0;
50  while (i < s1 and j < s2) {
51  if (kmer_p_i[i] == kmer_e[j]) {
52  ++interseccion;
53  ++i;
54  ++j;
55  }
56  else if (kmer_p_i[i] < kmer_e[j]) ++i;
57  else ++j;
58  }
59  // por el principio de inclusión exclusión, el cardinal de la unión de kmer_p_i
60  // y kmer_e es el cardinal de kmer_p_i más el cardinal de kmer_e menos el
61  // cardinal de la intersección de ambos. Aunque los kmer pueden contener
62  // elementos repetidos, cuando aplicamos el PIE se considera como si fueran
63  // distintos, ya que en la práctica, la unión coge el máximo de apariciones de
64  // cada elemento entre los dos kmer y la intersección, el mínimo
65  double card_union = s1 + s2 - interseccion;
66  return (1 - (interseccion/card_union))*100; //retorna la distancia
67 }
